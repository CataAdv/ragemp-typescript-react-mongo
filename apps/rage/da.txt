const mongoose = require('mongoose');

mongoose.connect('mongodb://ragemp_user:parola_sigura_aici@localhost:27017/ragemp', {
  useNewUrlParser: true,
  useUnifiedTopology: true
});

// Schema exemplu pentru user
const userSchema = new mongoose.Schema({
  socialClub: { type: String, required: true, unique: true },
  username: String,
  money: { type: Number, default: 5000 },
  bank: { type: Number, default: 10000 },
  level: { type: Number, default: 1 },
  position: {
    x: Number,
    y: Number,
    z: Number
  },
  lastIP: String,
  hwid: String,
  createdAt: { type: Date, default: Date.now },
  lastLogin: Date
});

const User = mongoose.model('User', userSchema);
